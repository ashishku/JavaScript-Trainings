  • Getter and setters 
    var o = {
    // An ordinary data property
    data_prop: value,
  
    // An accessor property defined as a pair of functions
    get accessor_prop()      { /* function body here */ },  // If getter is missing value is write only
    set accessor_prop(value) { /* function body here */ }   // If setter is missing value is read only
  };
