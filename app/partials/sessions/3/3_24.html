  • Methods
    • String Methods
      • search()
        Returns charecter position of text found(-1 if not found) 
          "JavaScript".search(/script/i);
      • replace()
        search and replace text 
          // No matter how it is capitalized, replace it with the correct capitalization
  text.replace(/javascript/gi, "JavaScript");
      • match()
        It takes a regular expression as its only argument and returns an array that contains the results of the match 
          "1 plus 2 equals 3".match(/\d+/g)    // returns ["1", "2", "3"]
    • RegExp Object
      • exec()
        If it find a match, it returns an array just like the array returned by the match(). 
          var pattern = /Java/g;
  var text = "JavaScript syntax is similar to Java!";
  var result;
  while((result = pattern.exec(text)) != null) {
    console.log(result);
  }
      • test()
        It takes a string and returns true if the string contains a match for the regular expression 
          var pattern = /java/i;
  pattern.test("JavaScript"); // Returns true
