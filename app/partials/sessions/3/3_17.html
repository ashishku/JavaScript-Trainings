  function max(/* ... */) {
    var max = Number.NEGATIVE_INFINITY;
    // Loop through the arguments, looking for, and remembering, the biggest.
    for(var i = 0; i < arguments.length; i++)
      if (arguments[i] > max) max = arguments[i];
    // Return the biggest
    return max;
  }
